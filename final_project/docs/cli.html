182    def op_seven(self, warehouses: Warehouses) -> None:
183        """Print Items at Warehouse"""
184        print("which warehouse would you like to see: ")
185        name = sys.stdin.readline()[:-1]
186
187        # print(warehouses.getWarehouse(name))
188
189        table = Table(title=name)
190
191        table.add_column("Item", justify="left",

194        table.add_column("Item Count", justify="left",
195                         style="cyan", no_wrap=True)
196
197        ware = warehouses.get_warehouse(name)

199            raise ValueError(f"Warehouse {name} does not exist..")
200        item_counts: Dict[str, int] = ware.cli_items()
201
202        for key, value in item_counts.items():

204            table.add_row(key, str(value))

206        self.console.print(table)

208    def op_eight(self, warehouses: Warehouses) -> None:
209        """Change Item Price"""
210        print("What is the item you would like to change: ")
211        item_name = sys.stdin.readline()[:-1]
212        print("What should the new price be: ")
213        new_price = sys.stdin.readline()[:-1]
214
215        found_item = False
216        # Iterate through all warehouses
217        for warehouse in warehouses.list_warehouses():
218            # Look for the item in this warehouse
219            item = next((item for item in warehouse.items if
220                         item.name == item_name), None)
221
222            if item:
223                # If the item is found, update its price
224                # Directly update the item's price
225                item.price = float(new_price)
226                print(f"The price of '{item_name}' has"
227                      f" been updated to {new_price}.")
228                found_item = True
229                break  # Exit the loop after updating the item
230
231        if not found_item:
232            raise ValueError(f"Item '{item_name}' not found in any warehouse.")
